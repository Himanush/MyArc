public static void plusMinus(List<Integer> arr) {
    // Write your code here
    float p =0,n=0,z =0;
    float len = arr.size();
    for(int num: arr){
        if(num<0){
            n=n+1;
        }else if(num>0){
            p=p+1;
        }else{
            z=z+1;
        }
    }
    System.out.println(p/len);
    System.out.println(n/len);
    System.out.println(z/len);
}
